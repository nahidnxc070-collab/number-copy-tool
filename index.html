function generateNumbers() { const base = document.getElementById('baseNumber').value; const total = 100; const list = document.getElementById('list'); list.innerHTML = ''; try { let baseNum = BigInt(base); // এখানে BigInt ব্যবহার করলাম for (let i = 0; i < total; i++) { const number = (baseNum + BigInt(i)).toString(); const div = document.createElement('div'); div.className = 'number-item'; div.innerHTML = `${number} `; list.appendChild(div); } document.getElementById('total').textContent = total; } catch (e) { alert("অনুগ্রহ করে শুধুমাত্র সংখ্যাসূচক ইনপুট দিন।"); } }
